\subsection{Interfaz de ColaPrioridad}

  \textbf{par\'ametros formales}\hangindent=2\parindent\\
  \parbox{1.7cm}{\textbf{g\'eneros}} $\alpha$\\
  \parbox[t]{1.7cm}{\textbf{funci\'on}}\parbox[t]{\textwidth-2\parindent-1.7cm}{%
    \InterfazFuncion{$\puntito$ < $\puntito$ }{\In{a}{$\alpha$}, \In{b}{$\alpha$}}{$\alpha$}
    [$true$]
    {$res \igobs a < b$}
    [$O(a < b)$]
    [Funcion menor de $\alpha$'s]
  }

\textbf{se explica con}: \tadNombre{Cola de Prioridad($\alpha$), Iterador Cola de Prioridades($\alpha$)}.

\textbf{g\'eneros}: \TipoVariable{colaPrioridad}, \TipoVariable{itcolaPrioridad}.


\Title{Operaciones b\'asicas de ColaPrioridad}

\InterfazFuncion{Vacia}{}{colaPrioridad}%
[true]   % Pre
{$res$ $\igobs$ $vacia$} %Post
[$O(1)$]   % Complejidad
[Genera una nueva cola de prioridades vac\'ia]  % Comentarios

\InterfazFuncion{EsVacia?}{\In{c}{colaPrioridad}}{bool}
{$res$ $\igobs$ vacia?($c$)}
[$O(1)$]
[Devuelve verdadero si la cola est\'a vac\'ia]

\InterfazFuncion{Pr\'oximo}{\In{c}{colaPrioridad}}{itcolaPrioridad)}
[$\neg$ vacia?($c$)]  
{$res$ $\igobs$ proximo($c$)}  % VER TEMA DE ALIASING 
[$O(1)$]
[Devuelve el pr\'oximo de la cola, sin eliminarlo]

\InterfazFuncion{Encolar}{\Inout{cola}{colaPrioridad($\alpha$)}, \In{e}{$\alpha$}}{}%
[true]   % Pre
{$res$ $\igobs$ encolar($e$, $cola$)} %Post
[O(log($\#$($cola$)))]   % Complejidad
[Encola el elemento en la cola de prioridad]  % Comentarios

\InterfazFuncion{Desencolar}{\Inout{c}{colaPrioridad}}{}
[$c$ $\igobs$ $c_0$ $\land$ $\neg$ vacia($c$) ]  
{$c$ $\igobs$ desencolar($c_0$)} 
[O(log($\#$($cola$)))]   % Complejidad
[Desencola el pr\'oximo de la cola]
