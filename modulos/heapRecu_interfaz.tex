\Title{Operaciones b\'asicas de ColaPrioridad}

\InterfazFuncion{Vacia}{}{colaPrioridad}%
[true]   % Pre
{$res$ $\igobs$ $vacia$} %Post
[$O(1)$]   % Complejidad
[Genera una nueva cola de prioridades vac\'ia]  % Comentarios

\InterfazFuncion{Encolar}{\Inout{cola}{colaPrioridad($\alpha$)}, \In{x}{$\alpha$}}{}%
[true]   % Pre
{$res$ $\igobs$ encolar($x$, $cola$)} %Post
[O(log($\#$($cola$)))]   % Complejidad
[Encola el elemento en la cola de prioridad]  % Comentarios
